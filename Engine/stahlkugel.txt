#Head
m = 1
rho_stahl = 7850
pi = 3.1415
v=0
cw=0.45
rho=1.29
h=2
EVer=0
verlust=0.05
g=9.81
delta_t=0.01
t=0

#Program
V=m / rho_stahl
r=(3 * V / (4 * pi))^(1/3)
A=pi*r*r
b=v*v*0.5*cw*rho*A
if (h-r<=0) EVer=EVer+verlust*1/2*m*v^2
if (h-r<=0) v=sqrt((1-verlust)*v*v)
if (h-r<=0) h=r
if (v>=0) b=0-b
F=m*(0-g)+b

//Standart-Programmablauf
a=F/m
v=v+a*delta_t
h=h+v*delta_t
t=t+delta_t
